题目描述：
把只包含因子2、3和5的数称作丑数（Ugly Number）。例如6、8都是丑数，但14不是，因为它包含因子7。
习惯上我们把1当做是第一个丑数。求按从小到大的顺序的第N个丑数。
输入：
输入包括一个整数N(1<=N<=1500)。
输出：
可能有多组测试数据，对于每组数据，
输出第N个丑数。
样例输入：
3
样例输出：
3
#include <stdio.h>
#include <stdlib.h>
int min( int a, int b, int c ){
        int min;
        min = (a<b)? a:b;
        min = (min<c)? min:c;
        return min;
}
int getUglyNumber( int len ){
        int *uglyNums = (int*)malloc( len*sizeof(int) );
        uglyNums[0] = 1;
        int next = 1;
        int *p2 = uglyNums;
        int *p3 = uglyNums;
        int *p5 = uglyNums;
        while( next<len ){
                uglyNums[next] = min(*p2*2, *p3*3, *p5*5 );
                while( uglyNums[next] >= *p2*2) p2++;
                while( uglyNums[next] >= *p3*3) p3++;
                while( uglyNums[next] >= *p5*5) p5++;
                next++;
        }
        int res = uglyNums[len-1];
        free(uglyNums);
        return res;
}
int main(){
    int n;
        while( scanf("%d",&n) != EOF ){
                printf("%d\n",getUglyNumber(n) );
        }
    return 0;
}
