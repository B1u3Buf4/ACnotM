A+B

问题描述 :

给定两个整数A和B，其表示形式是：从个位开始，每三位数用逗号","隔开。
现在请计算A+B的结果，并以正常形式输出。

输入:

输入包含多组数据数据，每组数据占一行，由两个整数A和B组成（-10^9 < A,B < 10^9）。

输出:

输入包含多组数据数据，每组数据占一行，由两个整数A和B组成（-10^9 < A,B < 10^9）。

样例输入:

-234,567,890 123,456,789
1,234 2,345,678

样例输出:

-111111101
2346912

#include<iostream>
#include<string>
using namespace std;
char a[20], b[20];

int main()
{
	while(cin >> a >> b)
	{
		int i, j, l;
		l = strlen(a);
		for( i=0; i < l; i++ )
		{
			if(a[i] == ',')
			{
				for( j=i; j < l; j++ )
					a[j] = a[j+1];
				l--;
				i--;
			}
		}
		l = strlen(b);
		for( i=0; i < l; i++ )
		{
			if(b[i] == ',')
			{
				for( j=i; j < l; j++ )
					b[j] = b[j+1];
				l--;
				i--;
			}
		}
		int aa, bb;
		aa = atoi(a); //将字符串转换成整形；
		bb = atoi(b);
		cout << aa+bb << endl;
	}
	return 0;
}
