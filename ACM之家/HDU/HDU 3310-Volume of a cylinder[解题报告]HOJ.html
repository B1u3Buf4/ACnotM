Volume of a cylinder

问题描述 :

One cylinder across another cylinder at right angles, tell the volume of the intersection of the cylinders.

输入:

Multiple test cases, the number of them T is given in the very first line.
For each test case, two integers r1 and r2, meaning the radius of the two cylinders. ( 1 < r1, r2 < 80 )

输出:

Multiple test cases, the number of them T is given in the very first line.
For each test case, two integers r1 and r2, meaning the radius of the two cylinders. ( 1 < r1, r2 < 80 )

样例输入:

1
4 4

样例输出:

341.33

#include <iostream>
#include <stdio.h>
#include <set>
#include <map>
#include <vector>
#include <math.h>
#include <algorithm>
#include <string>
#include <string.h>
#include <stdlib.h>
using namespace std;
const int N = 1000000;

double smallArea( double r1, double r2, double	x){
	return (sqrt(r1*r1 - x*x)*sqrt(r2*r2 - x*x));
}

double getArea( double r1, double r2){
	double area = 0;
	double dx = r1/(N*1.0);
	double x;
	for(int i=0; i<N; i++){
		x = i*dx*1.0+0.5*dx;
		area += smallArea( r1, r2, x );
	}
	return area*dx*8;
}

int main(){
	int t;
	double r1, r2, tmp;
	double area;
	
	 cin >> t;
		while( t--){
			cin >> r1 >> r2;
			if( r1 > r2 ){
				tmp = r1; 
				r1 = r2;
				r2 = tmp;
			}
			area = getArea( r1, r2 );
			printf("%.2lf\n", area);
		}
	
	return 0;
}
