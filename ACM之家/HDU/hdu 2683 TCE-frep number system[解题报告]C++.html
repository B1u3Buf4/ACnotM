Output

 

每组case包含三个整数分别表示白色，红色和黑色线段的个数。

#include <iostream>


using namespace std;


const int N = 10005;


int c[N];


int Lowbit(int m)


{


    return m&(-m);


}


void Change(int k,int m,int num)


{


   while(k<=num){


    c[k] += m;


    k += Lowbit(k);


   }


}


int Getsum(int k)


{


    int sum = 0;


    while(k>0){


        sum += c[k];


        k -= Lowbit(k);


    }


    return sum;


}


 


int main()


{


    int n, m;


    int from, to;


    while(scanf("%d%d", &n, &m) == 2)


    {


        memset(c, 0, sizeof(c));


        for(int i = 0; i < m; i++)


        {


            scanf("%d%d", &from, &to);


            Change(from, 1, n);


            Change(to + 1, -1, n);


        }


        int w = 0, r = 0, b = 0;


        for(int i = 1; i <= n; i++)


        {


            int t = Getsum(i) % 3;


            if(t == 0)


                w++;　


            else if(t == 1)


                r++;


            else if(t == 2)


                b++;


        }


        printf("%d %d %d\n", w, r, b);


    }


    return 0;


}
