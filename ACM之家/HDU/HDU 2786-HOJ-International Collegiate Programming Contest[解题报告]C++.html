公里转化为英里

公里数用最少个斐波那契数表示

即42表示为34+8 而不是34+5+2+1

#include <stdio.h>

void calc_fib(int f[]);

int main()
{
    int i, f[32];
    int t, x, y;
    bool flag[32];

    calc_fib(f);
    scanf("%d", &t);
    while(t--)
    {
        for (i = 1; i < 22; i++)
            flag[i] = 0;
        y = 0;
        scanf("%d", &x);
        for (i = 21; i > 0; i--)
        {
            if (x >= f[i])
            {
                x -= f[i];
                flag[i] = 1;
            }
            if (x == 0)
                break;
        }
        for (i = 2; i < 22; i++)
        {
            y += flag[i] * f[i-1];
        }
        printf("%d\n", y);
    }

    return 0;
}

void calc_fib(int f[])
{
    int i;

    f[1] = 1, f[2] = 2;
    for (i = 3; i < 22; i++)
    {
        f[i] = f[i-1] + f[i-2];
    }
}
