概率算法求解π[随机算法]

coder经典问题围观1929次20 条评论

求π要先求面积，正方形的面积是可以直接求的。

可以在正方形内随机的取点，越多越好，最后找出落在园内的点个数。根据比值，可以近似求得面积。

#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<time.h>
int main()
{
int j;
for(j=0;j<10;j++) {
	int i,n,k;
	double x,y,f; k=0;
	srand( (unsigned)time(0));

	printf("%d\n",time(0));

	printf(" 输入次数：");
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		x=rand()/(double)RAND_MAX;
		y=rand()/(double)RAND_MAX;
		f=sqrt(1-x*x);
		if(y<=f)
			k++;
	}
	printf("pai=%5.8f\n",4.0*k/n);
	return 0;
}
}
